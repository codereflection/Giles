    def self.create_rake_task(mod)
      if mod.class == Class
        tasknames = Array.new

        if mod.const_defined?("TaskName")
          tasknames << eval("#{mod}::TaskName")
        else
         tasknames << mod.name.downcase
        end

        tasknames.flatten.each do |taskname|
          Albacore.create_task taskname, mod
        end
      end
    end
