  def attr_hash(*names)
    names.each do |n|
      self.class_eval("def \#{n}(*value)\nif value.nil? || value.empty?\ninstance_variable_get(\"@\#{n}\")\nelse\ninstance_variable_set(\"@\#{n}\", value[0])\nend\nend\n", __FILE__, __LINE__)
     self.send :define_method, "#{n}=" do |value|
        instance_variable_set("@#{n}", value)
      end
    end
  end
