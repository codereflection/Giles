  def execute()
    fail_with_message 'Output File cannot be empty' if @output_file.nil?
    return if @output_file.nil?
        
    clean_directories_names
    remove zip_name

    ZipFile.open(zip_name, 'w')  do |zipfile|
      zip_directory(zipfile)
      zip_additional(zipfile)
    end
  end
