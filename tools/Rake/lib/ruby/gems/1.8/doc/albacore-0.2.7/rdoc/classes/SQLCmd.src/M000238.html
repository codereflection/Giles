  def get_authentication_params
    integratedParam = "-E" if @trusted_connection
    if ((!(@username.nil?)) and (!(@password.nil?)))
      integratedParam = build_parameter("U", @username) + " " + build_parameter("P", @password)
    end
    integratedParam
  end
