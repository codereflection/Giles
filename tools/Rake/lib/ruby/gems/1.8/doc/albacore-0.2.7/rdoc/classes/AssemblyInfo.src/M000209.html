  def build_using_statements
    @namespaces = [] if @namespaces.nil?
    
    @namespaces << "System.Reflection"
    @namespaces << "System.Runtime.InteropServices"
    @namespaces.uniq!
    
    ns = []
    @namespaces.each do |n|
      ns << @lang_engine.build_using_statement(n)
    end
    
    ns
  end
