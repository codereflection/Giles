  def execute()
    fail_with_message 'No base dir' if @from_dir.nil?
    fail_with_message 'No output dir' if @to_dir.nil?
    
    OutputBuilder.output_to(@to_dir, @from_dir)  do |o|
        @directories.each { |f| o.dir f }
        @files.each { |f| o.file *f }
        @erbs.each { |f| o.erb *f }
    end
  end
