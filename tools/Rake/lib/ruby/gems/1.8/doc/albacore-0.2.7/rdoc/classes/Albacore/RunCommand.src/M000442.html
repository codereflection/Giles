    def get_command(params)
      executable = @command
      unless command.nil?
        executable = File.expand_path(@command) if File.exists?(@command)
      end
      cmd = "\"#{executable}\""
      cmd +=" #{params.join(' ')}" if params.length > 0
      cmd
    end
