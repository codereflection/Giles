  def self.create_task(taskname, taskclass)
    # this style of creating tasks is not really what i
    # want to do. but it's necessary for ruby 1.8.6
    # because that version doesn't support the foo do |*args, &block|
    # block signature. it supports *args, but not &block.
    # so that limitation is worked around with string eval
    Object.class_eval("def \#{taskname}(name=:\#{taskname}, *args, &configblock)\ntask name, *args do |t, task_args|\nobj = \#{taskclass}.new\nobj.load_config_by_task_name(name) if obj.respond_to?(:load_config_by_task_name)\n\nif !configblock.nil?\ncase configblock.arity\nwhen 0\nconfigblock.call\nwhen 1\nconfigblock.call(obj)\nwhen 2\nconfigblock.call(obj, task_args)\nend\nend\n\nobj.execute if obj.respond_to?(:execute)\nend\nend\n", __FILE__, __LINE__)
  end
